56 void GeoElement::Jacobian(const MatrixDouble &gradx, MatrixDouble &jac, MatrixDouble &axes, double &detjac, MatrixDouble &jacinv) const {
57  int nrows = gradx.rows ();
58  int ncols = gradx.cols();
59  int dim = jac.rows ();
60  jac.resize(dim, dim);
61  axes.resize(dim, 3);
62  jacinv.resize(dim, dim);
63  jac.setZero();
64 
65  switch (dim) {
66  case 1:
67  {
68  axes.resize(dim, 3);
69  VecDouble v_1(3);
70  v_1.setZero();
71 
72  for (int i = 0; i < nrows; i++) {
73  v_1[i] = gradx(i, 0);
74  }
75 
76  double norm_v_1 = 0.;
77  for (int i = 0; i < nrows; i++) {
78  norm_v_1 += v_1[i] * v_1[i];
79  }
80 
81  norm_v_1 = sqrt(norm_v_1);
82  jac(0, 0) = norm_v_1;
83  detjac = norm_v_1;
84  jacinv(0, 0) = 1.0 / detjac;
85 
86  detjac = fabs(detjac);
87 
88  for (int i = 0; i < 3; i++) {
89  axes(0, i) = v_1[i] / norm_v_1;
90  }
91  }
92  break;
93  case 2:
94  {
95  VecDouble v_1(3), v_2(3);
96  VecDouble v_1_til(3), v_2_til(3);
97  v_1.setZero();
98  v_2.setZero();
99  v_1_til.setZero();
100  v_2_til.setZero();
101 
102  for (int i = 0; i < nrows; i++) {
103  v_1[i] = gradx(i, 0);
104  v_2[i] = gradx(i, 1);
105  }
106 
107  double norm_v_1_til = 0.0;
108  double norm_v_2_til = 0.0;
109  double v_1_dot_v_2 = 0.0;
110 
111  for (int i = 0; i < 3; i++) {
112  norm_v_1_til += v_1[i] * v_1[i];
113  v_1_dot_v_2 += v_1[i] * v_2[i];
114  }
115  norm_v_1_til = sqrt(norm_v_1_til);
116 
117  for (int i = 0; i < 3; i++) {
118  v_1_til[i] = v_1[i] / norm_v_1_til;
119  v_2_til[i] = v_2[i] - v_1_dot_v_2 * v_1_til[i] / norm_v_1_til;
120  norm_v_2_til += v_2_til[i] * v_2_til[i];
121  }
122  norm_v_2_til = sqrt(norm_v_2_til);
123 
124 
125  jac(0, 0) = norm_v_1_til;
126  jac(0, 1) = v_1_dot_v_2 / norm_v_1_til;
127  jac(1, 1) = norm_v_2_til;
128 
129  detjac = jac(0, 0) * jac(1, 1) - jac(1, 0) * jac(0, 1);
130 
131  jacinv(0, 0) = +jac(1, 1) / detjac;
132  jacinv(1, 1) = +jac(0, 0) / detjac;
133  jacinv(0, 1) = -jac(0, 1) / detjac;
134  jacinv(1, 0) = -jac(1, 0) / detjac;
135 
136  detjac = fabs(detjac);
137 
138  for (int i = 0; i < 3; i++) {
139  v_2_til[i] /= norm_v_2_til;
140  axes(0, i) = v_1_til[i];
141  axes(1, i) = v_2_til[i];
142  }
143  }
144  break;
145  case 3:
146  {
147  axes.resize(dim, 3);
148 
149  for (int i = 0; i < nrows; i++) {
150  jac(i, 0) = gradx(i, 0);
151  jac(i, 1) = gradx(i, 1);
152  jac(i, 2) = gradx(i, 2);
153  }
154 
155  detjac -= jac(0, 2) * jac(1, 1) * jac(2, 0); //- a02 a11 a20
156  detjac += jac(0, 1) * jac(1, 2) * jac(2, 0); //+ a01 a12 a20
157  detjac += jac(0, 2) * jac(1, 0) * jac(2, 1); //+ a02 a10 a21
158  detjac -= jac(0, 0) * jac(1, 2) * jac(2, 1); //- a00 a12 a21
159  detjac -= jac(0, 1) * jac(1, 0) * jac(2, 2); //- a01 a10 a22
160  detjac += jac(0, 0) * jac(1, 1) * jac(2, 2); //+ a00 a11 a22
161 
162  jacinv(0, 0) = (-jac(1, 2) * jac(2, 1) + jac(1, 1) * jac(2, 2)) / detjac; //-a12 a21 + a11 a22
163  jacinv(0, 1) = (jac(0, 2) * jac(2, 1) - jac(0, 1) * jac(2, 2)) / detjac; //a02 a21 - a01 a22
164  jacinv(0, 2) = (-jac(0, 2) * jac(1, 1) + jac(0, 1) * jac(1, 2)) / detjac; //-a02 a11 + a01 a12
165  jacinv(1, 0) = (jac(1, 2) * jac(2, 0) - jac(1, 0) * jac(2, 2)) / detjac; //a12 a20 - a10 a22
166  jacinv(1, 1) = (-jac(0, 2) * jac(2, 0) + jac(0, 0) * jac(2, 2)) / detjac; //-a02 a20 + a00 a22
167  jacinv(1, 2) = (jac(0, 2) * jac(1, 0) - jac(0, 0) * jac(1, 2)) / detjac; //a02 a10 - a00 a12
168  jacinv(2, 0) = (-jac(1, 1) * jac(2, 0) + jac(1, 0) * jac(2, 1)) / detjac; //-a11 a20 + a10 a21
169  jacinv(2, 1) = (jac(0, 1) * jac(2, 0) - jac(0, 0) * jac(2, 1)) / detjac; //a01 a20 - a00 a21
170  jacinv(2, 2) = (-jac(0, 1) * jac(1, 0) + jac(0, 0) * jac(1, 1)) / detjac; //-a01 a10 + a00 a11
171 
172  detjac = fabs(detjac);
173 
174  axes.setZero();
175  axes(0, 0) = 1.0;
176  axes(1, 1) = 1.0;
177  axes(2, 2) = 1.0;
178  }
179  break;

184 static bool CheckQuadConsistency(const VecInt &one, const VecInt &two)
185 {
186  // same orientation
187  for(int rot = 0; rot<4; rot++)
188  {
189  bool check = true;
190  for(int i=0; i<4; i++)
191  {
192  if(one[i] != two[(i+rot)%4]) {
193  check = false;
194  break;
195  }
196  }
197  if(check) return true;
198  }
199  // oposite rotation
200  for(int rot = 0; rot<4; rot++)
201  {
202  bool check = true;
203  for(int i=0; i<4; i++)
204  {
205  if(one[i] != two[(4-i+rot)%4]) {
206  check = false;
207  break;
208  }
209  }
210  if(check) return true;
211  }
212  return false;
213 }

215 int GeoElement::WhichSide(VecInt &SideNodeIds) const {
216  int64_t cap = SideNodeIds.size();
217  std::set<int> neighids;
218  neighids.insert(&SideNodeIds[0], &SideNodeIds[0]+cap);
219 
220  int nums = NSides();
221  for (int side = 0; side < nums; side++) {
222  if (NSideNodes(side) == cap) {
223  if(SideIsUndefined(side)) DebugStop();
224  std::set<int> mynodes;
225  for(int n=0; n<cap; n++) mynodes.insert(SideNodeIndex(side, n));
226  if(mynodes == neighids)
227  {
228  if(cap == 4)
229  {
230  VecInt mynodes(4);
231  for(int n=0; n<4; n++) mynodes[n] = SideNodeIndex(side, n);
232  if(CheckQuadConsistency(SideNodeIds, mynodes) == false) DebugStop();
233  }
234  return side;
235  }
236  }
237  }
238  return -1;
