144 static GeoMesh *BuildGMesh(MElementType eltype)
145 {
146  GeoMesh *gmesh = new GeoMesh();
147  MatrixDouble cornerco;
148  MasterCo(eltype, cornerco);
149  int nnodes = cornerco.cols();
150  int dim = cornerco.rows();
151  gmesh->SetNumNodes(nnodes);
152  gmesh->SetDimension(dim);
153  VecInt nodes(nnodes);
154  for(int i=0; i<nnodes; i++)
155  {
156  VecDouble co(cornerco.rows());
157  for(int ir=0; ir<cornerco.rows(); ir++) co[ir] = cornerco(ir,i);
158  GeoNode &node = gmesh->Node(i);
159  node.SetCo(co);
160  nodes[i] = i;
161  }
162  int index = 0;
163  AddElement(eltype, nodes, index, gmesh);
164  GeoElement *gel = gmesh->Element(0);
165  int nsides = gel->NSides();
166  for(int is = 0; is<nsides-1; is++)
167  {
168  int nsidenodes = gel->NSideNodes(is);
169  VecInt sidenodes(nsidenodes);
170  for(int i=0; i<nsidenodes; i++) sidenodes[i] = gel->SideNodeIndex(is, i);
171  AddElement(gel->Type(is), sidenodes, is+1, gmesh);
172  }
173  gmesh->BuildConnectivity();
174  return gmesh;
175 }

206 void VerifyShape(CompMesh *cmesh, int side)
207 {
208  CAPTURE(side);
209  CompElement *elroot = cmesh->GetElement(0);
210  GeoElement *gelroot = elroot->GetGeoElement();
211  CompElement *celside = cmesh->GetElement(side+1);
212  GeoElement *gelside = celside->GetGeoElement();
213  int dim = cmesh->GetGeoMesh()->Dimension();
214  // build the correspondence between the DOF of celside and DOF of root
215  // compute the first shape for each DOF
216  std::map<int,int> rootDOF;
217  int rootndof = elroot->NDOF();
218  int sidendof = celside->NDOF();
219  VecInt firstshapeRoot(rootndof),firstshapeSide(sidendof);
220  firstshapeRoot.setZero();
221  firstshapeSide.setZero();
222  for(int dof=0; dof<elroot->NDOF(); dof++)
223  {
224  int dofindex = elroot->GetDOFIndex(dof);
225  rootDOF[dofindex] = dof;
226  if(dof < elroot->NDOF()-1) firstshapeRoot[dof+1] = firstshapeRoot[dof]+elroot->NShapeFunctions(dof);
227  }
228  for(int dof=0; dof<celside->NDOF(); dof++)
229  {
230  int dofindex = celside->GetDOFIndex(dof);
231  if (rootDOF.find(dofindex) == rootDOF.end()) {
232  DebugStop();
233  }
234  if(dof < celside->NDOF()-1) firstshapeSide[dof+1] = firstshapeSide[dof]+celside->NShapeFunctions(dof);
235  }
236  MElementType sidetype = gelroot->Type(side);
237  std::unique_ptr<IntRule> intrule = GetRule(sidetype);
238  int npoints = intrule->NPoints();
239  VecDouble point(intrule->Dimension());
240  int root_nshape = elroot->NShapeFunctions();
241  int side_nshape = celside->NShapeFunctions();
242  VecDouble phiRoot(root_nshape);
243  VecDouble phiSide(side_nshape);
244  MatrixDouble dphiroot(gelroot->Dimension(),root_nshape);
245  MatrixDouble dphiside(gelside->Dimension(),side_nshape);
246  double weight;
247  for (int ip = 0; ip < npoints; ip++) {
248  intrule->Point(ip, point, weight);
249  VecDouble qsiroot(dim);
250  gelside->X(point,qsiroot);
251  elroot->ShapeFunctions(qsiroot, phiRoot, dphiroot);
252  celside->ShapeFunctions(point, phiSide, dphiside);
253  int ndof = celside->NDOF();
254  int count = 0;
255  for(int idf = 0; idf<ndof; idf++)
256  {
257  if(celside->GetDOF(idf).GetNShape() == 0) continue;
258  int dofindex = celside->GetDOFIndex(idf);
259  int rootlocdof = rootDOF[dofindex];
260  int sidepos = firstshapeSide[idf];
261  int rootpos = firstshapeRoot[rootlocdof];
262  REQUIRE_THAT(phiSide[sidepos], Catch::Matchers::WithinAbs(phiRoot[rootpos],1.e-8));
263  }
264  }
265 }
