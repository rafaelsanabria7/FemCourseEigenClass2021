71 void GeoMesh::BuildConnectivity() {
72  // for each node : an element index / side connected to the node
73  // if the value == -1, the datastructure hasn't been initialized yet
74  int numnodes = NumNodes();
75  VecInt sides(numnodes);
76  sides.setConstant(-1);
77  VecInt vetor(numnodes);
78  vetor.setConstant(-1);
79 
80 
81  int64_t nelem = this->NumElements();
82  int64_t iel = 0;
83 
84  // reset the connectivity
85  for (iel = 0; iel < nelem; iel++) {
86  GeoElement *gel = Elements[iel];
87  if (!gel) continue;
88  int nsides = gel->NSides();
89  for(int is=0; is<nsides; is++)
90  {
91  GeoElementSide gelside(gel,is);
92  gel->SetNeighbour(is, gelside);
93  }
94  }
95 
96  // set the connectivity along the nodes
97  for (iel = 0; iel < nelem; iel++) {
98  GeoElement *gel = Elements[iel];
99  if (!gel) continue;
100  int ncor = gel->NCornerNodes();
101  int in = 0;
102  for (in = 0; in < ncor; in++) {
103  int64_t nodeindex = gel->NodeIndex(in);
104  if (vetor[nodeindex] == -1) {
105  vetor[nodeindex] = iel;
106  sides[nodeindex] = in;
107 
108  } else {
109  GeoElementSide one(gel, in);
110  GeoElementSide two(Element(vetor[nodeindex]), sides[nodeindex]);
111 
112  GeoElementSide neighbour = one.Neighbour();
113  if (neighbour.Element() == 0) DebugStop();
114 
115  if (!two.IsNeighbour(one)) {
116  one.IsertConnectivity(two);
117  }
118  }
119  }
120  }
121  // at this point all neighbours along the nodes have been initialized
122 
123  // the line, triangle or quad neighbours are identified by the intersection of neighbours along nodes
124  for (iel = 0; iel < nelem; iel++) {
125  GeoElement *gel = Elements[iel];
126  if (!gel) continue;
127  int ncor = gel->NCornerNodes();
128  int nsides = gel->NSides();
129  int is;
130  for (is = ncor; is < nsides; is++) {
131  GeoElementSide gelside(gel, is);
132  GeoElementSide neigh = gelside.Neighbour();
133  if(gelside != neigh)
134  {
135  continue;
136  }
137  // the neighbours are the intersection of all neighbours along the nodes
138  std::vector<GeoElementSide> neighbours;
139  gelside.ComputeNeighbours(neighbours);
140  int64_t nneigh = neighbours.size();
141 // for(int in=0; in<nneigh; in++)
142 // {
143 // neighbours[in].Print(std::cout);
144 // }
145 // std::cout << std::endl;
146  for (int in = 0; in < nneigh; in++) {
147  gelside.IsertConnectivity(neighbours[in]);
148  }
149  }
150  }
151 }
